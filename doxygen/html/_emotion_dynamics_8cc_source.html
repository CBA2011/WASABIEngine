00142 EmotionDynamics::update(float _dt) {
00143     //cout << "EmotionDynamics::update: BEGIN" << endl;
00144     int yTensTemp = yTens;
00145     double tempPosY = sylast;
00146     bool yOk = positions2Reach->getYValid();
00147     int yPos2Reach = positions2Reach->getYPos2Reach();
00148     if (yOk) {
00149         tempPosY = sylast - yPos2Reach;
00150         //yTensTemp *= 3;
00151         if (tempPosY > 0){
00152             if (ySign > 0){
00153                 ySignChange = true;
00154                 ySign = 0;
00155             }
00156             else {
00157                 ySign = -1;
00158                 ySignChange = false;
00159             }
00160         }
00161         if (tempPosY < 0){
00162             if (ySign < 0){
00163                 ySignChange = true;
00164                 ySign = 0;
00165             }
00166             else {
00167                 ySign = 1;
00168                 ySignChange = false;
00169             }
00170         }
00171         if ((ySignChange == true) || (int)tempPosY == 0){// && yPos2Reach != 0){
00172             cout << "Position REACHED !!!!!!" << endl;
00173             ySign = 0;
00174             ySignChange = true;
00175             positions2Reach->setYValid(false);// = -101;
00176             positions2Reach->setPositions2Reach(positions2Reach->getXPos2Reach(), 0);
00177             vylast = 0;
00178             aylast = 0;
00179         }
00180     }
00181 
00182     // Fxt = -xTens * xPos, axt = Fx / mass --> axt = (-xTens * xPos) / mass
00183     double Fx = (-(xTens) * sxlast);
00184     double Fy = (-yTensTemp * tempPosY);
00185     //cout << "Fx = " << Fx << endl;
00186     axt = Fx / mass;
00187     //xcout << "-- axt = " << axt << endl;
00188     vxt = axt * dt + vxlast;
00189     sxt = vxt * dt + sxlast;
00190     if ((sxt > 0 && sxlast < 0) || (sxt < 0 && sxlast > 0)) {
00191         xPos = 0;
00192         sxt = sxlast = 0;
00193         vxlast = 0;
00194         axlast = 0;
00195     }
00196     else {
00197         xPos = (int)sxt;
00198         vxlast = vxt;
00199         axlast = axt;
00200         sxlast = sxt;
00201     }
00202     ayt = Fy / mass;
00203     vyt = ayt * dt + vylast;
00204     syt = vyt * dt + sylast;
00205     syt += sxt * ((float)(slope) / 100) / mass;
00206 
00207     if (sxt > 100)
00208         sxt = 100;
00209     if (syt > 100)
00210         syt = 100;
00211     if (sxt < -100)
00212         sxt = -100;
00213     if (syt < -100)
00214         syt = -100;
00215 
00216     if (((syt > 0 && sylast < 0) || (syt < 0 && sylast > 0))
00217             && !(positions2Reach->getYValid())) {
00218         yPos = 0;
00219         syt = sylast = 0;
00220         vylast = 0;
00221         aylast = 0;
00222     }
00223     else {
00224         yPos = (int)syt;
00225         vylast = vyt;
00226         aylast = ayt;
00227         sylast = syt;
00228     }
00229 
00230     //cout << "xPos = " << xPos << ", yPos = " << yPos << endl;
00231 
00232     if (xPos < xReg && yPos < yReg && xPos > -(xReg) && yPos > -(yReg)){
00233         if (z > -100) {
00234             z -= ((float)(boredom) / 1000);
00235         }
00236     }
00237     else {
00238         z = 0;
00239     }
00240     zPos = (int)z;
00241     // dominance update!
00242     if (dValue != dom_target){
00243         //     double Fdom = (-60 * (sdomlast - dom_target));
00244         //     adom = Fdom / mass;
00245         //     vdom = adom * dt + vdomlast;
00246         //     sdom = vdom * dt + sdomlast;
00247         //     dValue = (int)sdom;
00248         //     vdomlast = vdom;
00249         //     adomlast = adom;
00250         //    sdomlast = sdom;
00251         dValue = sdom = (int)dom_target;
00252     }
00253     else {
00254         dValue = (int)sdom;
00255         sdomlast = sdom;
00256         vdomlast = 0;
00257         adomlast = 0;
00258     }
00259     //cout << "EmotionDynamics::update: xPos = " << xPos << ", yPos = " << yPos << ", zPos = " << zPos << endl;
