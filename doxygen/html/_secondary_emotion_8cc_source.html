00074                 switch (returnIndex(keyword, "polygon_begin vertex lifetime baseIntensity type tokens_begin decayFunction")) {

00076                     cout << "SecondaryEmotion::SecondaryEmotion: loading polygon" << endl;

00092                         if (keyword == "vertex"){

00094                                 if (!(isline2 >> token1 && isline2 >> token2 && isline2 >> token3 && isline2 >> token4)){
00095                                     cerr << "SecondaryEmotion::loadFromFile: ERROR one of PAD or intensity missing!" << endl;
00096                                     cerr << "line: " << line << endl;

00107                             }

00111                         cout << "#######################" << endl;

00114                         returnValue = addPolygon(ap);
00115                     }
00116                     break;
00117                 case 2: // vertex
00118                     cerr << "SecondaryEmotion::loadFromFile: WARNING found vertex without prior polygon_begin! Ignoring.." << endl;
00119                     break;
00120                 case 3: // lifetime
00121                     cout << "lifetime" << endl;
00122                     if (!(isline >> token1)) {
00123                         cerr << "SecondaryEmotion::loadFromFile: ERROR no <double> lifetime found!" << endl;
00124                         break;
00125                     }

00128                         cerr << "SecondaryEmotion::loadFromFile: WARNING invalid lifetime " << doubleValue << endl;
00129                     }
00130                     break;
00131                 case 4: // baseIntensity
00132                     //cout << "intensity" << endl;
00133                     if (!(isline >> token1)) {
00134                         cerr << "SecondaryEmotion::loadFromFile: ERROR no <double> baseIntensity found!" << endl;
00135                         break;

00170                     }
00171                     break;
00172                 case 7: // decayFunction
00173                     //cout << "decayFunction" << endl;
00174                     if (!(isline >> token1)) {// && isline >> token2)) {

00176                         break;

00182                     else if (token1 == "LINEAR") {
00183                         setDecayFunction(LINEAR);
00184                     }
00185                     else if (token1 == "EXPONENTIAL") {
00186                         setDecayFunction(EXPONENTIAL);
00187                     }

00189                         setDecayFunction(COSINE);
00190                     }

00192                         cout << "SecondaryEmotion::loadFromFile: WARNING decayFunction " << token1 << " unknown!" << endl;
00193                     }
00194                     break;
00195                 default: // unknown keyword
00196                     cout << "SecondaryEmotion::loadFromFile: WARNING unknown keyword " << keyword << endl;
00197                 }
00198             }
00199         } // end if not '#'
